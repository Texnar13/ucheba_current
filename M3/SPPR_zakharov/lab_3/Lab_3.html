import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

#  
data = pd.read_csv('diabetes_dataset.csv')

#    smoking_history
grouped = data.groupby('smoking_history')['blood_glucose_level']

#     
stats = grouped.agg(mean='mean', median='median').reset_index()

#    Y
y_min = 120
y_max = 150

#    
categories = stats['smoking_history']
x = np.arange(len(categories))  #   
width = 0.35  #  

#  
fig, ax = plt.subplots(figsize=(10, 6))
ax.bar(x - width/2, stats['mean'], width, label='', color='g')
ax.bar(x + width/2, stats['median'], width, label='', color='c')

#    
ax.set_xticks(x)
ax.set_xticklabels(categories, rotation=45, ha='right')
ax.set_xlabel(' ')
ax.set_ylabel('   ')
ax.set_title('      ')
ax.legend()

#     Y
ax.set_ylim(y_min, y_max)

#  
plt.tight_layout()
plt.show()

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

#  
data = pd.read_csv('diabetes_dataset.csv')

#    
diabetes_data = data[data['diabetes'] == 1]

#    smoking_history     blood_glucose_level
grouped = diabetes_data.groupby('smoking_history')['bmi']
stats = grouped.agg(mean='mean', median='median').reset_index()

#     Y
y_min = 25  #    Y
y_max = 34  #    Y

#    Bar Chart
categories = stats['smoking_history']
x = np.arange(len(categories))  #   
width = 0.4  #  

#  Bar Chart
fig, ax = plt.subplots(figsize=(10, 6))
ax.bar(x - width/2, stats['mean'], width, label='', color='limegreen')
ax.bar(x + width/2, stats['median'], width, label='', color='cyan')

#    
ax.set_xticks(x)
ax.set_xticklabels(categories, rotation=45, ha='right')
ax.set_xlabel(' ')
ax.set_ylabel('   ')
ax.set_title('          ')
ax.legend()

#    Y
ax.set_ylim(y_min, y_max)

#     
ax.grid(axis='y', linestyle='--', alpha=0.7)

#  
plt.tight_layout()
plt.show()

import matplotlib.pyplot as plt
import numpy as np

#     X   
x1 = np.arange(0, 2, 0.01)  #   (  0  2)
x2 = np.arange(2, 4, 0.01)  #   (  2  5)
x3 = np.arange(4, 6, 0.01)  #   (  5  9)

#      
y1 = np.ones_like(x1) * 1  #  1   
y2 = np.ones_like(x2) * 2  #  2   
y3 = np.ones_like(x3) * 3  #  3   

#  Stacked Area Chart
plt.fill_between(x1, 0, y1, color='skyblue', label=' 1')  #  
plt.fill_between(x2, 0, y2, color='limegreen', label=' 2')  #  
plt.fill_between(x3, 0, y3, color='salmon', label=' 3')  #  

#  
plt.xlabel(' X')
plt.ylabel(' Y')
plt.title('Stacked Area Chart   ')
plt.legend(loc='upper left')

#  
plt.tight_layout()
plt.show()
